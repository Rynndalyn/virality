    <script>
        (function() {
            console.log("Initializing Virality Index...");
            
            const canvas = document.getElementById('canvas');
            const ctx = canvas.getContext('2d');
            
            let canvasWidth = 0;
            let canvasHeight = 0;
            
            // VIBRANT COLOR PALETTES
            const DORMANT_PALETTE = [
                [70, 90, 120],   // muted indigo
                [90, 80, 105],   // muted purple
                [80, 100, 90],   // muted teal
                [110, 80, 85],   // muted rose
                [75, 95, 105],   // muted blue
                [100, 75, 90],   // muted mauve
            ];
            
            const VIRAL_PALETTE = [
                [255, 100, 80],   // vibrant coral
                [80, 220, 180],   // electric mint
                [255, 180, 70],   // golden orange
                [70, 200, 255],   // bright cyan
                [255, 120, 160],  // hot pink
                [160, 130, 255],  // vibrant lavender
            ];
            
            let gameState = {
                influence: 0,
                currentMode: 'nonViral',
                momentum: 0,
                lastClickTime: 0,
                viralStreak: 0
            };
            
            class Node {
                constructor(x, y, paletteIdx) {
                    this.x = x;
                    this.y = y;
                    this.paletteIdx = paletteIdx;
                    this.dormantColor = [...DORMANT_PALETTE[paletteIdx % DORMANT_PALETTE.length]];
                    this.viralColor = [...VIRAL_PALETTE[paletteIdx % VIRAL_PALETTE.length]];
                    this.currentColor = [...this.dormantColor];
                    this.isConverted = false;
                    this.progress = 0;
                    this.vx = (Math.random() - 0.5) * 0.2;
                    this.vy = (Math.random() - 0.5) * 0.2;
                    this.size = 3 + Math.random() * 2.5;
                }
                
                update() {
                    this.x += this.vx;
                    this.y += this.vy;
                    
                    if (this.x < -20) this.x = canvasWidth + 20;
                    if (this.x > canvasWidth + 20) this.x = -20;
                    if (this.y < -20) this.y = canvasHeight + 20;
                    if (this.y > canvasHeight + 20) this.y = -20;
                }
                
                applyConversion(intensity) {
                    if (!this.isConverted) {
                        this.progress += intensity;
                        if (this.progress >= 1) {
                            this.isConverted = true;
                            this.progress = 1;
                        }
                        
                        // Smooth color transition
                        const t = Math.pow(this.progress, 1.2);
                        this.currentColor[0] = Math.floor(this.dormantColor[0] * (1 - t) + this.viralColor[0] * t);
                        this.currentColor[1] = Math.floor(this.dormantColor[1] * (1 - t) + this.viralColor[1] * t);
                        this.currentColor[2] = Math.floor(this.dormantColor[2] * (1 - t) + this.viralColor[2] * t);
                        
                        return this.progress >= 1;
                    }
                    return false;
                }
                
                draw() {
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                    
                    const alpha = this.isConverted ? 0.95 : 0.7;
                    ctx.fillStyle = `rgba(${this.currentColor[0]}, ${this.currentColor[1]}, ${this.currentColor[2]}, ${alpha})`;
                    ctx.fill();
                    
                    // Glow for viral nodes
                    if (this.isConverted) {
                        ctx.shadowBlur = 10;
                        ctx.shadowColor = `rgba(${this.currentColor[0]}, ${this.currentColor[1]}, ${this.currentColor[2]}, 0.5)`;
                        ctx.fill();
                        ctx.shadowBlur = 0;
                    }
                    
                    ctx.strokeStyle = `rgba(255, 255, 255, ${this.isConverted ? 0.3 : 0.15})`;
                    ctx.lineWidth = 0.8;
                    ctx.stroke();
                    
                    // Progress ring
                    if (!this.isConverted && this.progress > 0) {
                        ctx.beginPath();
                        ctx.arc(this.x, this.y, this.size + 2.5, 0, Math.PI * 2 * this.progress);
                        ctx.strokeStyle = `rgba(255, 200, 100, 0.9)`;
                        ctx.lineWidth = 2;
                        ctx.stroke();
                    }
                }
            }
            
            let nodes = [];
            let ripples = [];
            let circles = [];
            let markers = [];
            
            function initNodes(count) {
                nodes = [];
                for (let i = 0; i < count; i++) {
                    nodes.push(new Node(
                        Math.random() * canvasWidth,
                        Math.random() * canvasHeight,
                        Math.floor(Math.random() * 8)
                    ));
                }
            }
            
            function resizeCanvas() {
                const container = canvas.parentElement;
                const maxWidth = Math.min(window.innerWidth - 24, 1000);
                const aspectRatio = 0.6;
                
                canvasWidth = maxWidth;
                canvasHeight = canvasWidth * aspectRatio;
                
                canvas.width = canvasWidth;
                canvas.height = canvasHeight;
                canvas.style.width = `${canvasWidth}px`;
                canvas.style.height = `${canvasHeight}px`;
                
                const nodeCount = Math.floor(canvasWidth * canvasHeight / 2000);
                initNodes(nodeCount);
            }
            
            function updateUI() {
                const converted = nodes.filter(n => n.isConverted).length;
                const total = nodes.length;
                const percent = total > 0 ? Math.floor((converted / total) * 100) : 0;
                
                document.getElementById('convertedCount').innerHTML = `${converted}<span style="font-size:11px; color:#8b8b9a;">/${total}</span>`;
                document.getElementById('reachPercent').innerHTML = `${percent}%`;
                document.getElementById('totalScore').innerHTML = gameState.influence;
                
                const momentumEl = document.getElementById('momentumValue');
                momentumEl.innerHTML = gameState.momentum;
            }
            
            function showFloatingPoints(x, y, points, isViral) {
                const pointsDiv = document.createElement('div');
                pointsDiv.className = 'floating-points';
                pointsDiv.textContent = `+${points}`;
                pointsDiv.style.left = `${x + canvas.getBoundingClientRect().left}px`;
                pointsDiv.style.top = `${y + canvas.getBoundingClientRect().top - 15}px`;
                pointsDiv.style.color = isViral ? '#ffb87a' : '#ffb87a';
                document.body.appendChild(pointsDiv);
                setTimeout(() => pointsDiv.remove(), 600);
            }
            
            function processEngagement(x, y) {
                const now = Date.now();
                if (now - gameState.lastClickTime < 900) {
                    gameState.momentum++;
                    if (gameState.currentMode === 'viral') {
                        gameState.viralStreak++;
                    }
                } else {
                    gameState.momentum = 1;
                    gameState.viralStreak = gameState.currentMode === 'viral' ? 1 : 0;
                }
                gameState.lastClickTime = now;
                
                // Dynamic radius
                let radius = gameState.currentMode === 'viral' ? 85 : 45;
                if (gameState.currentMode === 'viral') {
                    const bonus = Math.min(90, Math.floor(gameState.viralStreak / 2) * 7);
                    radius += bonus;
                }
                
                // Conversion intensity
                let intensity = gameState.currentMode === 'viral' ? 
                    0.4 + Math.min(0.35, Math.floor(gameState.viralStreak / 4) * 0.05) : 0.18;
                
                let newlyConverted = 0;
                
                nodes.forEach(node => {
                    const dx = node.x - x;
                    const dy = node.y - y;
                    const distance = Math.sqrt(dx * dx + dy * dy);
                    
                    if (distance < radius) {
                        const completed = node.applyConversion(intensity);
                        if (completed) newlyConverted++;
                    }
                });
                
                const pointsPerConversion = gameState.currentMode === 'viral' ? 5 : 2;
                const pointsEarned = newlyConverted * pointsPerConversion;
                gameState.influence += pointsEarned;
                
                // Visual effects - WHITE CIRCLES
                ripples.push({ x, y, r: 5, maxR: radius, alpha: 0.7 });
                circles.push({ x, y, radius: radius, alpha: 0.6, mode: gameState.currentMode });
                markers.push({ x, y, size: 3, alpha: 0.8 });
                
                if (pointsEarned > 0) {
                    showFloatingPoints(x, y, pointsEarned, gameState.currentMode === 'viral');
                }
                
                // Haptic feedback
                if (window.navigator && window.navigator.vibrate) {
                    window.navigator.vibrate(10);
                }
                
                updateUI();
            }
            
            // DRAWING FUNCTIONS - WHITE CIRCLE FRAMES
            function drawCircleFrames() {
                for (let i = circles.length - 1; i >= 0; i--) {
                    const c = circles[i];
                    
                    // Main outer ring - WHITE
                    ctx.beginPath();
                    ctx.arc(c.x, c.y, c.radius, 0, Math.PI * 2);
                    ctx.strokeStyle = `rgba(255, 255, 255, ${c.alpha * 0.8})`;
                    ctx.lineWidth = 2.5;
                    ctx.stroke();
                    
                    // Inner ring - WHITE with glow
                    ctx.beginPath();
                    ctx.arc(c.x, c.y, c.radius * 0.7, 0, Math.PI * 2);
                    ctx.strokeStyle = `rgba(255, 255, 255, ${c.alpha * 0.5})`;
                    ctx.lineWidth = 1.5;
                    ctx.stroke();
                    
                    // Add a subtle glow
                    ctx.shadowBlur = 8;
                    ctx.shadowColor = `rgba(255, 255, 255, ${c.alpha * 0.3})`;
                    ctx.beginPath();
                    ctx.arc(c.x, c.y, c.radius, 0, Math.PI * 2);
                    ctx.stroke();
                    ctx.shadowBlur = 0;
                    
                    c.alpha -= 0.012;
                    if (c.alpha <= 0) circles.splice(i, 1);
                }
            }
            
            function drawRipples() {
                for (let i = ripples.length - 1; i >= 0; i--) {
                    const r = ripples[i];
                    ctx.beginPath();
                    ctx.arc(r.x, r.y, r.r, 0, Math.PI * 2);
                    ctx.strokeStyle = `rgba(255, 220, 150, ${r.alpha})`;
                    ctx.lineWidth = 2;
                    ctx.stroke();
                    r.r += 3.8;
                    r.alpha -= 0.02;
                    if (r.r > r.maxR || r.alpha <= 0) ripples.splice(i, 1);
                }
            }
            
            function drawMarkers() {
                for (let i = markers.length - 1; i >= 0; i--) {
                    const m = markers[i];
                    ctx.beginPath();
                    ctx.arc(m.x, m.y, m.size, 0, Math.PI * 2);
                    ctx.fillStyle = `rgba(255, 220, 150, ${m.alpha * 0.7})`;
                    ctx.fill();
                    m.size += 1.8;
                    m.alpha -= 0.05;
                    if (m.alpha <= 0) markers.splice(i, 1);
                }
            }
            
            function drawBackground() {
                const gradient = ctx.createLinearGradient(0, 0, canvasWidth, canvasHeight);
                gradient.addColorStop(0, '#08080c');
                gradient.addColorStop(1, '#0c0c14');
                ctx.fillStyle = gradient;
                ctx.fillRect(0, 0, canvasWidth, canvasHeight);
            }
            
            function animate() {
                if (!canvasWidth || !canvasHeight) {
                    requestAnimationFrame(animate);
                    return;
                }
                
                nodes.forEach(node => node.update());
                
                drawBackground();
                nodes.forEach(node => node.draw());
                drawCircleFrames();
                drawRipples();
                drawMarkers();
                
                requestAnimationFrame(animate);
            }
            
            // MOBILE-FRIENDLY TOUCH HANDLING
            let isTouching = false;
            
            function handleTap(e) {
                e.preventDefault();
                e.stopPropagation();
                
                let clientX, clientY;
                
                if (e.touches) {
                    clientX = e.touches[0].clientX;
                    clientY = e.touches[0].clientY;
                } else {
                    clientX = e.clientX;
                    clientY = e.clientY;
                }
                
                const rect = canvas.getBoundingClientRect();
                if (!rect) return;
                
                // Calculate relative position
                let x = (clientX - rect.left) * (canvasWidth / rect.width);
                let y = (clientY - rect.top) * (canvasHeight / rect.height);
                
                // Boundary check
                if (x >= 0 && x <= canvasWidth && y >= 0 && y <= canvasHeight) {
                    processEngagement(x, y);
                }
            }
            
            // Touch events with proper handling
            canvas.addEventListener('touchstart', handleTap, { passive: false });
            canvas.addEventListener('touchmove', (e) => {
                e.preventDefault();
            }, { passive: false });
            canvas.addEventListener('touchend', (e) => {
                e.preventDefault();
            });
            canvas.addEventListener('mousedown', handleTap);
            canvas.addEventListener('contextmenu', (e) => e.preventDefault());
            
            // Button handlers
            document.getElementById('nonViralBtn').addEventListener('click', (e) => {
                gameState.currentMode = 'nonViral';
                gameState.viralStreak = 0;
                document.getElementById('nonViralBtn').classList.add('active');
                document.getElementById('viralBtn').classList.remove('active');
                updateUI();
            });
            
            document.getElementById('viralBtn').addEventListener('click', (e) => {
                gameState.currentMode = 'viral';
                document.getElementById('viralBtn').classList.add('active');
                document.getElementById('nonViralBtn').classList.remove('active');
                updateUI();
            });
            
            document.getElementById('resetBtn').addEventListener('click', (e) => {
                gameState = {
                    influence: 0,
                    currentMode: gameState.currentMode,
                    momentum: 0,
                    lastClickTime: 0,
                    viralStreak: 0
                };
                initNodes(Math.floor(canvasWidth * canvasHeight / 2000));
                ripples = [];
                circles = [];
                markers = [];
                updateUI();
            });
            
            // Keyboard support
            window.addEventListener('keydown', (e) => {
                if (e.key === '1') {
                    gameState.currentMode = 'nonViral';
                    gameState.viralStreak = 0;
                    document.getElementById('nonViralBtn').classList.add('active');
                    document.getElementById('viralBtn').classList.remove('active');
                    updateUI();
                    e.preventDefault();
                } else if (e.key === '2') {
                    gameState.currentMode = 'viral';
                    document.getElementById('viralBtn').classList.add('active');
                    document.getElementById('nonViralBtn').classList.remove('active');
                    updateUI();
                    e.preventDefault();
                } else if (e.key === 'r' || e.key === 'R') {
                    gameState = {
                        influence: 0,
                        currentMode: gameState.currentMode,
                        momentum: 0,
                        lastClickTime: 0,
                        viralStreak: 0
                    };
                    initNodes(Math.floor(canvasWidth * canvasHeight / 2000));
                    ripples = [];
                    circles = [];
                    markers = [];
                    updateUI();
                    e.preventDefault();
                }
            });
            
            // Hide touch hint after first interaction
            let hasInteracted = false;
            function hideHint() {
                if (!hasInteracted) {
                    const hint = document.getElementById('touchHint');
                    hint.style.opacity = '0';
                    setTimeout(() => {
                        if (hint) hint.style.display = 'none';
                    }, 500);
                    hasInteracted = true;
                }
            }
            
            canvas.addEventListener('touchstart', hideHint);
            canvas.addEventListener('mousedown', hideHint);
